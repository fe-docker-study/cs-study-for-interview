import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;
// 토마토가 모두 익는데 걸리는 최소 날짜 구하기
public class Main {
	static class Tomato{
		int x;
		int y;
		int rottenDay;
		public Tomato(int x, int y, int rottenDay) {
			this.x = x;
			this.y = y;
			this.rottenDay = rottenDay;
		}
	}
	public static void printMap(int[][] map, int m, int n) {
		for(int i=0; i<n; i++) {
			for(int j=0; j<m; j++) {
				System.out.print(map[i][j]+" ");
			}
			System.out.println();
		}
		System.out.println("-----------------");
	}
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int m = sc.nextInt();	// 가로 칸 수 
		int n = sc.nextInt();	// 세로 칸 수 
		int[][] map = new int[n][m];	// 1: 익음, 0: 익지 않음, -1: 없음
		int[] dx = {1,-1,0,0};
		int[] dy = {0,0,1,-1};
		
		int cnt = 0;	// 안익은 토마토 개수
		
		Queue<Tomato> q = new LinkedList<>();
		
		for(int i=0; i<n; i++) {
			for(int j=0; j<m; j++) {
				map[i][j] = sc.nextInt();
				if(map[i][j]==1) {
					q.add(new Tomato(i, j, 1));
				} else if(map[i][j]==0) {
					cnt++;
				}
			}
		}
		
		if(cnt==0) {
			System.out.println(0);
		}else {
			int days = 0;
			while(!q.isEmpty()) {
				if(cnt==0) break;
				Tomato tmt = q.remove();
				days = Math.max(days, tmt.rottenDay);
//				printMap(map,m,n);
				
				for(int i=0; i<4; i++) {
					int x = tmt.x+dx[i];
					int y = tmt.y+dy[i];
					if(x<0 || y<0 || x>=n || y>=m) continue;
					
					if(map[x][y]==0) {
						cnt--;
						map[x][y]=1;
						q.add(new Tomato(x, y, tmt.rottenDay+1));
					}
				}
//				System.out.println("cnt : "+cnt);
			}
			
			if(cnt!=0) {
				System.out.println(-1);
			}else {
				System.out.println(days);
			}
		}
		
	}
}
